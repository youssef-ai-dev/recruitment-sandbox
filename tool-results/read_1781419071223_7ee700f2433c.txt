  1159→        const lines = clean.split('\n')
  1160→            .map(l => l.replace(/^[-•*◦▪]\s*/, '').replace(/^\d+[.)]\s*/, '').trim())

  1169→        result.experienceLevel = extractExperience(clean);
  1170→
  1171→        // --- Detect section boundaries ---
  1172→        const sections = detectSections(lines);
  1173→
  1174→        // --- Extract skills from full text ---
  1175→        const lowerClean = clean.toLowerCase();
  1176→        result.hardSkills = extractHardSkills(lowerClean);
  1177→        result.softSkills = extractSoftSkills(lowerClean);
