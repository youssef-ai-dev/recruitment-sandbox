  1194→    function extractTitle(lines, fullText) {
  1195→        // Strategy 1: Look for role-like first line
  1196→        if (lines.length > 0) {
  1197→            const first = lines[0].replace(/[:\-–—]+$/, '').trim();
  1198→            if (first.length < 80) {
  1199→                const lower = first.toLowerCase();
  1200→                // Check if it contains role keywords
  1201→                if (ROLE_KEYWORDS.some(k => lower.includes(k))) {
  1202→                    return first;
  1203→                }
  1204→                // Check if first line is short and looks like a title
  1205→                if (first.length < 50 && !first.includes('.') && lines.length > 1) {
  1206→                    return first;
  1207→                }
  1208→            }
  1209→        }
  1210→
  1211→        // Strategy 2: Search for role pattern in text
  1212→        const rolePattern = /(?:looking for|seeking|hire|hiring)\s+(?:a|an)\s+(.+?)(?:\.|,|\n|$)/i;
  1213→        const match = fullText.match(rolePattern);
  1214→        if (match) {
  1215→            let candidate = match[1].trim();
  1216→            // Clean up
  1217→            candidate = candidate.replace(/\s+(?:to|who|that|for)\b.*$/i, '').trim();
  1218→            if (candidate.length < 60) return capitalizeTitle(candidate);
  1219→        }
  1220→
  1221→        // Strategy 3: Find a line with role keywords
  1222→        for (const line of lines.slice(0, 5)) {
  1223→            const lower = line.toLowerCase();
  1224→            const hasRole = ROLE_KEYWORDS.some(k => lower.includes(k));
  1225→            if (hasRole && line.length < 70) {
  1226→                return line.replace(/[:\-–—]+$/, '').trim();
  1227→            }
  1228→        }
  1229→
  1230→        return 'Untitled Role';
  1231→    }

  1639→    function simpleHash(str) {

  1641→        for (let i = 0; i < str.length; i++) {
